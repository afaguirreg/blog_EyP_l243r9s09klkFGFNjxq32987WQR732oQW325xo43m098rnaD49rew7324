#include <LiquidCrystal_PCF8574.h>

 

LiquidCrystal_PCF8574 lcd(0x27);

 

#define pin_A0 36

#define pin_A1 39

#define pin_A2 34

 

int ADC_0;

int ADC_1;

int ADC_2;

 

void setup() {

  // put your setup code here, to run once:

  lcd.begin(20, 4); // initialize the lcd

  delay(200);

  lcd.setBacklight(255);

  lcd.print("ADC_0:");

  lcd.setCursor(0, 1);

  lcd.print("ADC_1:");

  lcd.setCursor(0, 2);

  lcd.print("ADC_2:");

}

 

void loop() {

  // put your main code here, to run repeatedly:

  ADC_0 = analogRead(pin_A0);

  ADC_1 = analogRead(pin_A1);

  ADC_2 = analogRead(pin_A2);

 

  lcd.setCursor(7, 0);

  lcd.print(ADC_0);

  lcd.print("   ");

  lcd.setCursor(7, 1);

  lcd.print(ADC_1);

  lcd.print("   ");

  lcd.setCursor(7, 2);

  lcd.print(ADC_2);

  lcd.print("   ");

 

  delay(100);

}
