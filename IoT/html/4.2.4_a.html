// Include the libraries we need
#include <SPI.h>
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

Adafruit_SSD1306 display = Adafruit_SSD1306(128, 32, &Wire);

#define pin_I0 16
#define pin_I1 17
#define pin_I2 18
#define pin_I3 19
#define pin_I4 20
#define pin_I5 21
#define pin_I6 15
#define pin_I7 14

#define I0 digitalRead(pin_I0)
#define I1 digitalRead(pin_I1)
#define I2 digitalRead(pin_I2)
#define I3 digitalRead(pin_I3)
#define I4 digitalRead(pin_I4)
#define I5 digitalRead(pin_I5)
#define I6 digitalRead(pin_I6)
#define I7 digitalRead(pin_I7)

#define Q0 6
#define Q1 7
#define Q2 8
#define Q3 9
#define Q4 10
#define Q5 11
#define Q6 12
#define Q7 13

char texto[] = "I/O digitales";

void estados() {
  display.clearDisplay();
  display.setCursor(0,0);
  display.println(texto);
  display.print("I: ");
  display.write(I0*43 + 176);
  display.print(" ");
  display.write(I1*43 + 176);
  display.print(" ");
  display.write(I2*43 + 176);
  display.print(" ");
  display.write(I3*43 + 176);
  display.print(" ");
  display.write(I4*43 + 176);
  display.print(" ");
  display.write(I5*43 + 176);
  display.print(" ");
  display.write(I6*43 + 176);
  display.print(" ");
  display.write(I7*43 + 176);
  display.println("");
  display.println("   0 1 2 3 4 5 6 7");
  display.print("Q: ");
  display.write(digitalRead(Q0)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q1)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q2)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q3)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q4)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q5)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q6)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q7)*43 + 176);
  display.display();
}

void setup() {
  // put your setup code here, to run once:
  pinMode(pin_I0, INPUT);
  pinMode(pin_I1, INPUT);
  pinMode(pin_I2, INPUT);
  pinMode(pin_I3, INPUT);
  pinMode(pin_I4, INPUT);
  pinMode(pin_I5, INPUT);
  pinMode(pin_I6, INPUT);
  pinMode(pin_I7, INPUT);
  pinMode(Q0, OUTPUT);
  pinMode(Q1, OUTPUT);
  pinMode(Q2, OUTPUT);
  pinMode(Q3, OUTPUT);
  pinMode(Q4, OUTPUT);
  pinMode(Q5, OUTPUT);
  pinMode(Q6, OUTPUT);
  pinMode(Q7, OUTPUT);

  // SSD1306_SWITCHCAPVCC = generate display voltage from 3.3V internally
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C); // Address 0x3C for 128x32
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  //Activar página de código 437
  display.cp437(true);
  display.display();
  delay(2000);
}

void loop() {
  // put your main code here, to run repeatedly:
  estados();
  if (I0 && !I1)
    digitalWrite(Q0, HIGH);
  if (I1)
    digitalWrite(Q0, LOW);
}
