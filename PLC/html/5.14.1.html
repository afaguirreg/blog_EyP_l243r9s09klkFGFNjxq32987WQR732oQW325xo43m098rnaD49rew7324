//////////////////////////////// sketch section ////////////////////////////////
#include <Adafruit_SH110X.h>

Adafruit_SH1106G display = Adafruit_SH1106G(128, 64, &Wire);

#define pin_I0 4
#define pin_I1 5
#define pin_I2 6
#define pin_I3 7

#define I0 digitalRead(pin_I0)
#define I1 digitalRead(pin_I1)
#define I2 digitalRead(pin_I2)
#define I3 digitalRead(pin_I3)

#define Q0 19
#define Q1 18
#define Q2 10
#define Q3 3

char texto[] = "digital I/O";

void display_update() {
  display.clearDisplay();
  display.setCursor(0, 0);
  display.println(texto);
  display.println("");
  display.print("I: ");
  display.write(I0*43 + 176);
  display.print(" ");
  display.write(I1*43 + 176);
  display.print(" ");
  display.write(I2*43 + 176);
  display.print(" ");
  display.write(I3*43 + 176);
  display.println("");
  display.println("   0 1 2 3");
  display.print("Q: ");
  display.write(digitalRead(Q0)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q1)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q2)*43 + 176);
  display.print(" ");
  display.write(digitalRead(Q3)*43 + 176);
  display.display();
}

void sketch_setup()
{
  pinMode(pin_I0, INPUT);
  pinMode(pin_I1, INPUT);
  pinMode(pin_I2, INPUT);
  pinMode(pin_I3, INPUT);
  pinMode(Q0, OUTPUT);
  pinMode(Q1, OUTPUT);
  pinMode(Q2, OUTPUT);
  pinMode(Q3, OUTPUT);

  display.begin(0x3C, true);
  display.setTextSize(1);
  display.setTextColor(SH110X_WHITE);
  display.cp437(true);
  display.display();
  delay(2000);
}

void sketch_loop()
{
  display_update();
}
