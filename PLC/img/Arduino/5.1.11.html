#include <LiquidCrystal_PCF8574.h>

LiquidCrystal_PCF8574 lcd(0x27);

void sketch_setup()
{
  lcd.begin(20, 4);
  delay(200);
  lcd.setBacklight(255);
  lcd.print("ADC_0: ");
}

void sketch_loop()
{
  lcd.setCursor(7, 0);
  lcd.print(ADC_0);
  lcd.print(" ");
  delay(100);
}
